▶ 자바_변수 선언 방법

1) 정수 선언

char k = 20;
int a = 10;
		
2) 실수 선언

float b = 20.2f;
double c = 20.33d;
		
3) 문자 문자열 선언

char d = 'A'; (한 문자일때는 홀따옴표)
String e = "안녕하세요"; (두 문자 이상은 쌍따옴표)
		
System.out.println(a);
System.out.println(b);
System.out.println(c);
System.out.println(d);
System.out.println(e);
